1. Write an Assembly language program to add a number 5 times using loop

.model small
.stack 100h
.data
num dw 05h ;initialize the number to add
count dw 05h ;initialize the loop counter
result dw 00h;initialize the result to zero
.code
main proc
mov ax,@data
mov ds,ax;initialize data segment
mov cx,count;load loop counter
mov bx,num ;load number to add
add_loop:
add result, bx;add number to result
loop add_loop counter and loop if not zero
mov ah,4ch;return control to DOS INT 21H
main endp
end main

2. Write an ALP to print hello world using INT 21H

.model small
.data
msg db 10d, 13d,"hello world $"
.code
mov ax,@data
mov ds,ax
lea dx,msg
mov ah,09h
int 21h
mov ah,4ch
int 21h
end

5. Write an ALP that prints out 10 fibonacci serious.

.model small
.stack 100h
.data
fib db 0,1
msg"Fibonacci Numbers:$"
.code
main proc
mov ax,@data
mov ds,ax
;print message
mov ah,09h
lea dx,msg
int 21h
;initialize loop counter
mov cx,10
;calculate and print Fibonacci numbersL1:
mov al,[fib+1]
add al,[fib]
mov[fib],[fib+1]
mov[fib+1],al
;print current Fibonacci number
mov ah,02h
mov dl,al
add dl,30h
int 21h
;print space character
mov dl,''
int 21h
loop L1
;exit program
mov ah,4ch
int 21h
main endp
end main
